// معالج أحداث القائمة الجانبية
class SidebarMenuEventHandler(
    private val menuController: SidebarMenuController,
    private val quickAccessManager: QuickAccessManager,
    private val reportsManager: ReportsManager
) {
    
    fun onMenuItemClicked(item: MenuItem) {
        viewModelScope.launch {
            try {
                // تحديث حالة العنصر
                menuController.updateMenuItem(item.copy(isActive = true))
                
                // تسجيل الاستخدام
                if (item.id.startsWith("quick_")) {
                    quickAccessManager.updateShortcutUsage(item.id)
                }
                
                // تنفيذ الإجراء
                executeMenuAction(item.action)
                
            } catch (e: Exception) {
                handleError(e)
            }
        }
    }
    
    fun onQuickAccessAdded(item: QuickAccessItem) {
        viewModelScope.launch {
            quickAccessManager.addCustomShortcut(item)
                .onSuccess { refreshQuickAccess() }
                .onFailure { showError(it.message) }
        }
    }
    
    fun onReportGenerated(type: ReportType, filters: ReportFilters) {
        viewModelScope.launch {
            reportsManager.generateReport(type, filters, ExportFormat.PDF)
                .onSuccess { shareReport(it) }
                .onFailure { showError(it.message) }
        }
    }
    
    private suspend fun executeMenuAction(action: String) {
        when (action) {
            "nav_dashboard" -> navigateToDashboard()
            "nav_shipments" -> navigateToShipments()
            "nav_reports" -> navigateToReports()
            "nav_settings" -> navigateToSettings()
            "action_new_shipment" -> createNewShipment()
            "action_track_shipment" -> openTracking()
            "action_emergency" -> triggerEmergency()
            else -> handleCustomAction(action)
        }
    }
}

// محسن الأداء للقائمة الجانبية
class SidebarMenuOptimizer {
    
    private val menuCache = mutableMapOf<UserRole, SidebarMenuConfig>()
    private val quickAccessCache = mutableMapOf<String, List<QuickAccessItem>>()
    private val performanceMetrics = mutableMapOf<String, Long>()
    
    suspend fun getOptimizedMenuConfig(userRole: UserRole): SidebarMenuConfig {
        return menuCache[userRole] ?: run {
            val startTime = System.currentTimeMillis()
            val config = loadMenuConfiguration(userRole)
            menuCache[userRole] = config
            performanceMetrics["menu_load_${userRole.name}"] = System.currentTimeMillis() - startTime
            config
        }
    }
    
    suspend fun preloadQuickAccess(userRole: UserRole) {
        viewModelScope.launch {
            try {
                val items = quickAccessManager.getQuickAccessItems(userRole)
                quickAccessCache[userRole.name] = items
                // تحميل مسبق للأيقونات والبيانات
                preloadIcons(items.map { it.icon })
                preloadData(items.map { it.action })
            } catch (e: Exception) {
                logError("Failed to preload quick access", e)
            }
        }
    }
    
    fun optimizeMemoryUsage() {
        // تنظيف ذاكرة التخزين المؤقت غير المستخدمة
        val currentTime = System.currentTimeMillis()
        menuCache.entries.removeIf { (_, config) ->
            currentTime - config.lastAccessed > CACHE_EXPIRY_TIME
        }
        
        // تحسين حجم البيانات
        quickAccessCache.values.forEach { items ->
            items.forEach { item ->
                item.compressedData = compressData(item.rawData)
            }
        }
    }
    
    private fun preloadIcons(icons: List<String>) {
        // تحميل مسبق للأيقونات
        icons.forEach { icon ->
            imageLoader.preload(icon)
        }
    }
    
    private fun preloadData(actions: List<String>) {
        // تحميل مسبق للبيانات
        actions.forEach { action ->
            dataLoader.preload(action)
        }
    }
}
